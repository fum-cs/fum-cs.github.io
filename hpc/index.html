🔌 ۳. دسترسی برنامه‌نویسی (Ollama API)

برای اتصال از طریق کد پایتون (مناسب برای ساخت Agent، خودکارسازی، و افزونه Continue در VSCode):

from ollama import Client

# داخل دانشگاه
client = Client(host='http://172.21.129.16:11434')

# خارج دانشگاه با Tailscale
# client = Client(host='http://100.71.218.12:11434')

model = 'qwen3:8b'
prompt = "Explain the difference between supervised and unsupervised learning in one paragraph."

response = client.chat(model=model, messages=[
    {'role': 'user', 'content': prompt}
])

print(response['message']['content'])
